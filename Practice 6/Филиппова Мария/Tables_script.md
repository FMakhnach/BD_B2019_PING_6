DROP TABLE IF EXISTS Athletes, Olympiads, Teams, Events, Achievements;

CREATE TABLE Teams(
	country VARCHAR(30) NOT NULL,
	team_id VARCHAR(3) UNIQUE,
	population INT
);

CREATE TABLE Athletes(
	name VARCHAR(40),
	athlete_id VARCHAR(20) UNIQUE,
	country VARCHAR(3),
	birthday DATE,
	gender INT CHECK (gender IN (0,1)),
	FOREIGN KEY (country) REFERENCES Teams (team_id)
);

CREATE TABLE Olympiads(
	olympiad_id VARCHAR(7) UNIQUE,
	country VARCHAR(40) NOT NULL,
	city VARCHAR(40) NOT NULL,
	year INT,
	season INT CHECK (season IN (0,1)),
	team_id VARCHAR(3),
	FOREIGN KEY (team_id) REFERENCES Teams(team_id)
);

CREATE TABLE Events(
	event_id VARCHAR(7) UNIQUE,
	name VARCHAR(40),
	place VARCHAR(30),
	olympiad_id VARCHAR(7),
	type INT CHECK (type IN (0,1)),
	time TIMESTAMP,
	FOREIGN KEY (olympiad_id) REFERENCES Olympiads (olympiad_id)
);

CREATE TABLE Achievements(
	event_id VARCHAR(7),
	athlete_id VARCHAR(10),
	medal VARCHAR(10),
	FOREIGN KEY (event_id) REFERENCES Events (event_id),
	FOREIGN KEY (athlete_id) REFERENCES Athletes (athlete_id)
);
